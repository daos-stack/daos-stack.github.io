218 static inline int d_log_check(int flags)
219 {
220  int fac = flags & DLOG_FACMASK;
221  int lvl = flags & DLOG_PRIMASK;
222  int msk;
223 
224  if (!d_log_xst.tag) /* Log isn't open */
225  return 0;
226 
227  /*
228  * Immediately skip if facility is disabled and the log level is less
229  * severe than DLOG_ERR. Otherwise, the mask will be checked again
230  * below. Essentially all error messages should be logged from all
231  * facilities, except if the user specifies a mask > DLOG_ERR.
232  */
233  if (!d_log_xst.dlog_facs[fac].is_enabled && lvl < DLOG_ERR)
234  return 0;
235 
236  /* Use default facility if it is malformed */
237  if (fac >= d_log_xst.fac_cnt)
238  fac = 0;
239 
240  /*
241  * first, see if we can ignore the log messages because it is
242  * masked out. if debug messages are masked out, then we just
243  * directly compare levels. if debug messages are not masked,
244  * then we allow all non-debug messages and for debug messages we
245  * check to make sure the proper bit is on. [apps that don't use
246  * the debug bits just log with DLOG_DBG which has them all set]
247  */
248  msk = d_log_xst.dlog_facs[fac].fac_mask;
249  if (lvl >= DLOG_INFO) {
250  if (lvl < msk)
251  return 0; /* Skip it */
252  } else { /* debug clog message */
253  /*
254  * note: if (msk >= DLOG_INFO), then all the mask's debug bits
255  * are zero (meaning debugging messages are masked out). thus,
256  * for messages with the debug level we only have to do a bit
257  * test.
258  */
259  if ((lvl & msk) == 0) /* do we want this type of debug msg? */
260  return 0; /* Skip it */
261  }
262 
263  return lvl | fac;
264 }
