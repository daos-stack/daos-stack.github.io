212 typedef struct {

215  /*
216  * Type of the value in an iod can be either a single type that is
217  * always overwritten when updated, or it can be an array of EQUAL sized
218  * records where the record is updated atomically. Note that an akey can
219  * only support one type of value which is set on the first update. If
220  * user attempts to mix types in the same akey, the behavior is
221  * undefined, even after the object, key, or value is punched. If
222  * \a iod_type == DAOS_IOD_SINGLE, then iod_nr has to be 1, and
223  * \a iod_size would be the size of the single atomic value. The idx is
224  * ignored and the rx_nr is also required to be 1.
225  */
226  daos_iod_type_t iod_type;
