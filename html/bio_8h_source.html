244 /*
245  * Convert bio_sglist into d_sg_list_t, caller is responsible to
246  * call d_sgl_fini(sgl, false) to free iovs.
247  */
248 static inline int
249 bio_sgl_convert(struct bio_sglist *bsgl, d_sg_list_t *sgl, bool deduped_skip)
250 {
251  int i, rc;
252 
253  D_ASSERT(sgl != NULL);
254  D_ASSERT(bsgl != NULL);
255 
256  rc = d_sgl_init(sgl, bsgl->bs_nr_out);
257  if (rc != 0)
258  return rc;
259 
260  sgl->sg_nr_out = bsgl->bs_nr_out;
261 
262  for (i = 0; i < sgl->sg_nr_out; i++) {
263  struct bio_iov *biov = &bsgl->bs_iovs[i];
264  d_iov_t *iov = &sgl->sg_iovs[i];
265 
266  /* Skip bulk transfer for deduped extent */
267  if (biov->bi_addr.ba_dedup && deduped_skip)
268  iov->iov_buf = NULL;
269  else
270  iov->iov_buf = bio_iov2req_buf(biov);
271  iov->iov_len = bio_iov2req_len(biov);
272  iov->iov_buf_len = iov->iov_len;
273  }
274 
275  return 0;
276 }
