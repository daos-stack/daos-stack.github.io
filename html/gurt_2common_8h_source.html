129 /* Requires newptr and oldptr to be different variables. Otherwise
130  * there is no way to tell the difference between successful and
131  * failed realloc.
132  */
133 #define D_REALLOC_COMMON(newptr, oldptr, size, cnt) \
134  do { \
135  size_t _esz = (size_t)(size); \
136  size_t _sz = (size_t)(size) * (cnt); \
137  size_t _cnt = (size_t)(cnt); \
138  /* Compiler check to ensure type match */ \
139  __typeof__(newptr) optr = oldptr; \
140  D_ASSERT((void *)&(newptr) != &(oldptr)); \
141  /* \
142  * On Linux, realloc(p, 0) may return NULL after \
143  * freeing p. This behavior has proved tricky, as it is \
144  * easy to mistake the NULL return value for an error \
145  * and keep using the dangling p. We therefore allocate \
146  * a 1-byte object in this case, simulating the BSD \
147  * behavior. \
148  */ \
149  if (_sz == 0) \
150  _sz = 1; \
151  if (D_SHOULD_FAIL(d_fault_attr_mem)) \
152  (newptr) = NULL; \
153  else \
154  (newptr) = d_realloc(optr, _sz); \
155  if ((newptr) != NULL) { \
156  if (_cnt <= 1) \
157  D_DEBUG(DB_MEM, \
158  "realloc '" #newptr \
159  "': %zu at %p (old '" #oldptr \
160  "':%p).\n", \
161  _esz, (newptr), (oldptr)); \
162  else \
163  D_DEBUG(DB_MEM, \
164  "realloc '" #newptr \
165  "': %zu * '" #cnt \
166  "':%zu at %p (old '" #oldptr \
167  "':%p).\n", \
168  _esz, _cnt, (newptr), (oldptr));\
169  (oldptr) = NULL; \
170  break; \
171  } \
172  if (_cnt <= 1) \
173  D_ERROR("out of memory (tried to realloc " \
174  "'" #newptr "': size=%zu)\n", \
175  _esz); \
176  else \
177  D_ERROR("out of memory (tried to realloc " \
178  "'" #newptr "': size=%zu count=%zu)\n", \
