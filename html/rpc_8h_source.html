105 static inline int
106 daos_rpc_tag(int req_type, int tgt_idx)
107 {
108  switch (req_type) {
109  /* for normal IO request, send to the main service thread/context */
110  case DAOS_REQ_IO:
111  case DAOS_REQ_DTX:
112  return DAOS_IO_CTX_ID(tgt_idx);

117  case DAOS_REQ_CONT:
118  case DAOS_REQ_REBUILD:
119  case DAOS_REQ_IV:
120  case DAOS_REQ_BCAST:
121  case DAOS_REQ_SWIM:
122  return 0;
123  default:
124  D_ASSERTF(0, "bad req_type %d.\n", req_type);
125  return -DER_INVAL;

143 static inline int
144 daos_rpc_register(struct crt_proto_format *proto_fmt, uint32_t cli_count,
145  struct daos_rpc_handler *handlers, int mod_id)
146 {
147  uint32_t i;
148 
149  /* TODO: mod_in is unused */
150 
151  if (proto_fmt == NULL)
152  return 0;
153 
154  if (handlers != NULL) {
155  /* walk through the RPC list and fill with handlers */
156  for (i = 0; i < proto_fmt->cpf_count; i++) {
157  proto_fmt->cpf_prf[i].prf_hdlr =
158  handlers[i].dr_hdlr;
159  proto_fmt->cpf_prf[i].prf_co_ops =
160  handlers[i].dr_corpc_ops;
161  }
162  } else {
163  proto_fmt->cpf_count = cli_count;
164  }
165 
166  return crt_proto_register(proto_fmt);
167 }
