698 #define CRT_GEN_PROC_FIELD(r, ptr, seq) \
699  BOOST_PP_IF(BOOST_PP_EQUAL(CRT_ARRAY, CRT_GEN_GET_KIND(seq)), \
700  { \
701  uint64_t count = ptr->CRT_GEN_GET_NAME(seq).ca_count; \
702  CRT_GEN_GET_TYPE(seq)** e_ptrp = &ptr->CRT_GEN_GET_NAME(seq).ca_arrays; \
703  CRT_GEN_GET_TYPE(seq)* e_ptr = ptr->CRT_GEN_GET_NAME(seq).ca_arrays; \
704  int i; \
705  crt_proc_op_t proc_op; \
706  rc = crt_proc_get_op(proc, &proc_op); \
707  if (rc) \
708  D_GOTO(out, rc); \
709  /* process the count of array first */ \
710  rc = crt_proc_uint64_t(proc, &count); \
711  if (rc) \
712  D_GOTO(out, rc); \
713  ptr->CRT_GEN_GET_NAME(seq).ca_count = count; \
714  if (count == 0) { \
715  if (proc_op == CRT_PROC_DECODE) \
716  *e_ptrp = NULL; \
717  goto next_field_##r; \
718  } \
719  if (proc_op == CRT_PROC_DECODE) { \
720  D_ALLOC_ARRAY(e_ptr, (int)count); \
721  if (e_ptr == NULL) \
722  D_GOTO(out, rc = -DER_NOMEM); \
723  *e_ptrp = e_ptr; \
724  } \
725  /* process the elements of array */ \
726  for (i = 0; i < count; i++) { \
727  rc = CRT_GEN_GET_FUNC(seq)(proc, &e_ptr[i]); \
728  if (rc) \
729  D_GOTO(out, rc); \
730  } \
731  if (proc_op == CRT_PROC_FREE) \
732  D_FREE(e_ptr); \
733  } \
734  next_field_##r:, \
735  BOOST_PP_IF(BOOST_PP_EQUAL(CRT_RAW, CRT_GEN_GET_KIND(seq)), \
736  rc = crt_proc_memcpy(proc, &ptr->CRT_GEN_GET_NAME(seq), \
737  sizeof(CRT_GEN_GET_TYPE(seq))); \
738  if (rc) \
739  D_GOTO(out, rc);, \
740  rc = CRT_GEN_GET_FUNC(seq)(proc, &ptr->CRT_GEN_GET_NAME(seq)); \
741  if (rc) \
742  D_GOTO(out, rc); \

1059 #define CRT_TREE_MAX_RATIO (64)

1062 /*
1063  * Calculate the tree topology. Can only be called on the server side.
1064  *
1065  * \param[in] tree_type tree type
1066  * \param[in] branch_ratio branch ratio, be ignored for CRT_TREE_FLAT.
1067  * for KNOMIAL tree or KARY tree, the valid value
1068  * should within the range of
1069  * [CRT_TREE_MIN_RATIO, CRT_TREE_MAX_RATIO], or
1070  * will be treated as invalid parameter.
1071  *
1072  * \return tree topology value on success,
1073  * negative value if error.
1074  */
1075 static inline int
1076 crt_tree_topo(enum crt_tree_type tree_type, uint32_t branch_ratio)
1077 {
1078  if (tree_type < CRT_TREE_MIN || tree_type > CRT_TREE_MAX)
1079  return -DER_INVAL;
1080 
1081  return (tree_type << CRT_TREE_TYPE_SHIFT) |
1082  (branch_ratio & ((1U << CRT_TREE_TYPE_SHIFT) - 1));
1083 }
