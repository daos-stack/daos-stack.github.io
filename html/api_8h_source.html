1087 /*
1088  * Calculate the tree topology. Can only be called on the server side.
1089  *
1090  * \param[in] tree_type tree type
1091  * \param[in] branch_ratio branch ratio, be ignored for CRT_TREE_FLAT.
1092  * for KNOMIAL tree or KARY tree, the valid value
1093  * should within the range of
1094  * [CRT_TREE_MIN_RATIO, CRT_TREE_MAX_RATIO], or
1095  * will be treated as invalid parameter.
1096  *
1097  * \return tree topology value on success,
1098  * negative value if error.
1099  */
1100 static inline int
1101 crt_tree_topo(enum crt_tree_type tree_type, uint32_t branch_ratio)
1102 {
1103  if (tree_type < CRT_TREE_MIN || tree_type > CRT_TREE_MAX)
1104  return -DER_INVAL;
1105 
1106  return (tree_type << CRT_TREE_TYPE_SHIFT) |
1107  (branch_ratio & ((1U << CRT_TREE_TYPE_SHIFT) - 1));
1108 }
